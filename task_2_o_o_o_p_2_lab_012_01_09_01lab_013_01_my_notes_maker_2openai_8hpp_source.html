142inline void Session::setMultiformPart(const std::pair<std::string, std::string>& fieldfield_and_filepath, const std::map<std::string, std::string>& fields) {
143 // https://curl.se/libcurl/c/curl_mime_init.html
144 if (curl_) {
145 if (mime_form_ != nullptr) {
146 curl_mime_free(mime_form_);
147 mime_form_ = nullptr;
148 }
149 curl_mimepart *field = nullptr;
150
151 mime_form_ = curl_mime_init(curl_);
152
153 field = curl_mime_addpart(mime_form_);
154 curl_mime_name(field, fieldfield_and_filepath.first.c_str());
155 curl_mime_filedata(field, fieldfield_and_filepath.second.c_str());
156
157 for (const auto &field_pair : fields) {
158 field = curl_mime_addpart(mime_form_);
159 curl_mime_name(field, field_pair.first.c_str());
160 curl_mime_data(field, field_pair.second.c_str(), CURL_ZERO_TERMINATED);
161 }
162
163 curl_easy_setopt(curl_, CURLOPT_MIMEPOST, mime_form_);
164 }
165}

189inline Response Session::makeRequest(const std::string& contentType) {
190 std::lock_guard<std::mutex> lock(mutex_request_);
191
192 struct curl_slist* headers = NULL;
193 if (!contentType.empty()) {
194 headers = curl_slist_append(headers, std::string{"Content-Type: " + contentType}.c_str());
195 if (contentType == "multipart/form-data") {
196 headers = curl_slist_append(headers, "Expect:");
197 }
198 }
199 headers = curl_slist_append(headers, std::string{"Authorization: Bearer " + token_}.c_str());
200 if (!organization_.empty()) {
201 headers = curl_slist_append(headers, std::string{"OpenAI-Organization: " + organization_}.c_str());
202 }
203 curl_easy_setopt(curl_, CURLOPT_HTTPHEADER, headers);
204 curl_easy_setopt(curl_, CURLOPT_URL, url_.c_str());
205
206 std::string response_string;
207 std::string header_string;
208 curl_easy_setopt(curl_, CURLOPT_WRITEFUNCTION, writeFunction);
209 curl_easy_setopt(curl_, CURLOPT_WRITEDATA, &response_string);
210 curl_easy_setopt(curl_, CURLOPT_HEADERDATA, &header_string);
211
212 res_ = curl_easy_perform(curl_);
213
214 bool is_error = false;
215 std::string error_msg{};
216 if(res_ != CURLE_OK) {
217 is_error = true;
218 error_msg = "OpenAI curl_easy_perform() failed: " + std::string{curl_easy_strerror(res_)};
219 if (throw_exception_) {
220 throw std::runtime_error(error_msg);
221 }
222 else {
223 std::cerr << error_msg << '\n';
224 }
225 }
226
227 return { response_string, is_error, error_msg };
228}

230inline std::string Session::easyEscape(const std::string& text) {
231 char *encoded_output = curl_easy_escape(curl_, text.c_str(), static_cast<int>(text.length()));
232 const auto str = std::string{ encoded_output };
233 curl_free(encoded_output);
234 return str;
235}
