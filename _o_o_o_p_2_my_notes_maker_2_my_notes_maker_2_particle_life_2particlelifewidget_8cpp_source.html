81{
82 int screenWidth = 1480;
83 int screenHeight = 1480;
84
85 for (auto a : firstGroup) {

87 double fy = 0;
88
89 for (auto b : secondGroup) {

91 int dy = a->y() - b->y();
92 double d = sqrt(dx * dx + dy * dy);
93
94 if (d > 0 && d < radius) {
95 double F = g / d;
96 fx += (F * dx);
97 fy += (F * dy);
