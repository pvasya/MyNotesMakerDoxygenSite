148void ParticleLifeWidget::animate(QList<Particle *> &firstGroup, QList<Particle *> &secondGroup, double g, int radius)
149{
150 int screenWidth = 1480;
151 int screenHeight = 1480;
152
153 for (auto it_a = firstGroup.begin(); it_a != firstGroup.end(); ++it_a) {
154 double fx = 0;
155 double fy = 0;
156 Particle *a = *it_a;
157
158 for (auto it_b = secondGroup.begin(); it_b != secondGroup.end(); ++it_b) {
159 Particle *b = *it_b;
160 int dx = a->x() - b->x();
161 int dy = a->y() - b->y();
162 double d = sqrt(dx * dx + dy * dy);
163
164 if (d > 0 && d < radius) {
165 double F = g / d;
166 fx += (F * dx);
167 fy += (F * dy);
168 }
169 }
170 a->setVX((a->getVX() + fx) * 0.5);
171 a->setVY((a->getVY() + fy) * 0.5);
172 //a->moveBy(a->getVX(), a->getVY());
173
174
175 if (a->x() <= 10 || a->x() >= screenWidth) {
176 a->setVX(-1*a->getVX());
177 }
178 if (a->y() <=10 || a->y() >= screenHeight) {
179 a->setVY(-1*a->getVY());
