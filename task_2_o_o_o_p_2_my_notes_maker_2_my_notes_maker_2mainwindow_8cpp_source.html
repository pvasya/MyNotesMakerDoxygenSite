213void MainWindow::find(QString text) {
214 QTextCursor currentCursor = ui->textEdit->textCursor();
215 QTextCursor cursor(ui->textEdit->document());
216
217 QList<QTextCursor> occurrences;
218 cursor = ui->textEdit->document()->find(text);
219 while (!cursor.isNull() && !cursor.atEnd()+1) {
220 occurrences.append(cursor);
221 cursor = ui->textEdit->document()->find(text, cursor);
222 }
223
224 if (!occurrences.isEmpty()) {
225 bool foundCurrentCursor = false;
226 for (const QTextCursor& foundCursor : occurrences) {
227 if (foundCursor.position() == currentCursor.position()) {
228 foundCurrentCursor = true;
229 continue;
230 }
231 if (foundCurrentCursor) {
232 ui->textEdit->setTextCursor(foundCursor);
233 return;
234 }
235 }
236 ui->textEdit->setTextCursor(occurrences.first());
237 }
238}
